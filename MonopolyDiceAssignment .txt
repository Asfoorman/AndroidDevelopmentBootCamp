import java.util.Randomfun main(args: Array<String>) {            printStatements()    }fun generateRandomNumber(): Int {    var random = Random()        var min = 1    var max = 6        val randomNum = random.nextInt(max + 1 - min) + min            return randomNum}fun printStatements() {            var counter = 0        do  {                var die1: Int = rollDice()        var die2: Int = rollDice()            println("User rolls " + die1 + " & " + die2)        if (die1 == die2) {                        println("DOUBLES!")            println("Move " + (die1+die2) + " Spaces and roll again getting...")            counter++        } else {             println("Move " + (die1+die2) + " Spaces and stop")        }                if (counter == 3) {            println("GO TO JAIL --->")            break;        }           } while(die1 == die2) //end of Do While Loop    }fun rollDice(): Int {        val dice = generateRandomNumber()        return dice}
